using namespace std;
void adding(int n1, int n2, int* A1, int* A2)
{
    if (n1 > n2)
    {
        int* result = new int[n1 + 1];
        result[0] = n1;
        int i(n1), j(n2),k(n1-n2);
        while (j >= 0)
        {
            result[i] = A1[i - 1] + A2[j - 1];
            i--; j--;
        }
        for (int h = 1; h < k + 1; h++)
        {
            result[h] = A1[h-1];
        }
        for (int h = 0; h < n1 + 1; h++)
        {
            cout << result[h]<<" ";
        }
        delete[] result;
    }
    else if (n1 < n2)
    {
        int* result = new int[n2 + 1];
        result[0] = n2;
        int i(n1), j(n2), k(n2 - n1);
        while (i >= 0)
        {
            result[j] = A1[i - 1] + A2[j - 1];
            i--; j--;
        }
        for (int h = 1; h < k + 1; h++)
        {
            result[h] = A2[h-1];
        }
        for (int h = 0; h < n2 + 1; h++)
        {
            cout << result[h]<<" ";
        }
        delete[] result;
    }
    else if(n1==n2)
    {
        int* result = new int[n1 + 1];
        result[0] = n1; 
        for (int i = n1; i > 0; i--)
        {
            result[i] = A1[i-1] + A2[i-1];
        }
        for (int h = 0; h < n1 + 1; h++)
        {
            cout << result[h]<<" ";
        }
        delete[] result;
    }
}
